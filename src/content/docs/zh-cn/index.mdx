import { CardGrid, Card } from "@astrojs/starlight/components"

<CardGrid>
  <Card title="持久化与可追溯" icon="open-book">
    状态与结果落盘保存（SQLite），每次执行都有完整记录；失败后可重试、可重跑、可回放，避免“一次性脚本”的不可控。
  </Card>
  <Card title="实时可观测与回放" icon="rocket">
    执行过程实时输出日志与事件；输入、输出、文件与产物都可查看/下载，定位问题更快，也更容易复现与审计。
  </Card>
  <Card title="隔离执行（执行器 + 沙盒容器）" icon="document">
    推荐生产把“编排与状态”与“执行与隔离”拆开：控制面负责调度与持久化，执行器在沙盒容器内运行并回传日志与产物。
  </Card>
  <Card title="可组合的工作流图" icon="puzzle">
    用步骤化编排把任务拆小：输入输出清晰、依赖关系明确、产物可交付；适合把零散脚本演进成长期可维护的自动化系统。
  </Card>
</CardGrid>
